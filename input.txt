void bubbleSort ?int arr[], int n?
{
    int n;
    int arr[10];
    int i;
    int j;
    int temp;
    i := 0;
    j := 0;
    temp := 0;
    
    whilst (i < n - 1) {
		j := i + 1;
        whilst (j < n - 1) {
            when (arr[j] > arr[j + 1]) {
                temp := arr[i];
                arr[i] := arr[j];
                arr[j] := temp;
            }
            j := j + 1;
        }
        i := i + 1;
    }
}
